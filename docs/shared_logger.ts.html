import pino, { Logger, LoggerOptions } from 'pino';

const pinoPrettifyOptions = {
  colorize: true,
  levelFirst: true,
  translateTime: true,
};

/**
 * Initiierung des Loggers mit prettyPrint Optionen
 */
export const logger: Logger = pino({
  prettyPrint: {
    colorize: true,
    levelFirst: true,
    translateTime: true,
  },
});
